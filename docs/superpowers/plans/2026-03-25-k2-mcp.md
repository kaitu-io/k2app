**Goal:** Build a standalone Go MCP server binary (`k2-mcp`) that exposes k2 tunnel operations and Kaitu account management as 8 MCP tools for customer AI agents.

**Architecture:** Standalone Go binary communicating via stdio (MCP protocol). Two HTTP targets: Center API (auth, servers, plans, orders) and k2 daemon localhost:1777 (connect, disconnect, status). Session tokens persist to `~/.kaitu/mcp-session.json`.

**Tech Stack:** Go 1.25+, `github.com/modelcontextprotocol/go-sdk/mcp` v1.4.1+, `net/http` stdlib

```go
package main

import (
	"bytes"
	"encoding/json"
	"errors"
	"fmt"
	"net/http"
	"sync"
	"time"
)

// CenterError represents a non-zero error code from the Center API.
type CenterError struct {
	Code    int    `json:"code"`
	Message string `json:"message"`
}

func (e *CenterError) Error() string {
	return fmt.Sprintf("center api error %d: %s", e.Code, e.Message)
}

// centerResponse is the JSON envelope returned by all Center API endpoints.
// Center API always returns HTTP 200; errors are indicated by non-zero code.
type centerResponse struct {
	Code    int             `json:"code"`
	Message string          `json:"message,omitempty"`
	Data    json.RawMessage `json:"data,omitempty"`
}

// CenterClient is an HTTP client for the Kaitu Center API.
type CenterClient struct {
	BaseURL string // e.g. "https://api.kaitu.io"

	mu    sync.RWMutex
	token string
	udid  string
	http  *http.Client
}

func NewCenterClient(baseURL string) *CenterClient {
	return &CenterClient{
		BaseURL: baseURL,
		http:    &http.Client{Timeout: 15 * time.Second},
	}
}

func (c *CenterClient) SetToken(token string) {
	c.mu.Lock()
	c.token = token
	c.mu.Unlock()
}

func (c *CenterClient) Token() string {
	c.mu.RLock()
	defer c.mu.RUnlock()
	return c.token
}

func (c *CenterClient) SetUDID(udid string) {
	c.mu.Lock()
	c.udid = udid
	c.mu.Unlock()
}

// Get performs a GET request and decodes the Center API JSON envelope.
// result receives the unmarshaled "data" field. Pass nil to ignore data.
func (c *CenterClient) Get(path string, result any) error {
	req, err := http.NewRequest(http.MethodGet, c.BaseURL+path, nil)
	if err != nil {
		return fmt.Errorf("create request: %w", err)
	}
	return c.do(req, result)
}

// Post performs a POST request with a JSON body and decodes the response.
func (c *CenterClient) Post(path string, body any, result any) error {
	data, err := json.Marshal(body)
	if err != nil {
		return fmt.Errorf("marshal body: %w", err)
	}
	req, err := http.NewRequest(http.MethodPost, c.BaseURL+path, bytes.NewReader(data))
	if err != nil {
		return fmt.Errorf("create request: %w", err)
	}
	req.Header.Set("Content-Type", "application/json")
	return c.do(req, result)
}

func (c *CenterClient) do(req *http.Request, result any) error {
	c.mu.RLock()
	token := c.token
	udid := c.udid
	c.mu.RUnlock()

	if token != "" {
		req.Header.Set("Authorization", "Bearer "+token)
	}
	if udid != "" {
		req.Header.Set("X-UDID", udid)
	}

	resp, err := c.http.Do(req)
	if err != nil {
		return fmt.Errorf("cannot reach Kaitu servers, check network: %w", err)
	}
	defer resp.Body.Close()

	var envelope centerResponse
	if err := json.NewDecoder(resp.Body).Decode(&envelope); err != nil {
		return fmt.Errorf("decode response: %w", err)
	}

	if envelope.Code != 0 {
		return &CenterError{Code: envelope.Code, Message: envelope.Message}
	}

	if result != nil && len(envelope.Data) > 0 {
		if err := json.Unmarshal(envelope.Data, result); err != nil {
			return fmt.Errorf("decode data: %w", err)
		}
	}
	return nil
}
```

```go
package main

import (
	"bytes"
	"encoding/json"
	"fmt"
	"net/http"
	"time"
)

// DaemonStatus represents the parsed status from the k2 daemon.
type DaemonStatus struct {
	State          string        `json:"state"`
	ConnectedAt    string        `json:"connected_at,omitempty"`
	UptimeSeconds  int           `json:"uptime_seconds,omitempty"`
	Config         *DaemonConfig `json:"config,omitempty"`
	Error          *DaemonError  `json:"error,omitempty"`
}

type DaemonConfig struct {
	Server string `json:"server"`
}

type DaemonError struct {
	Code    int    `json:"code"`
	Message string `json:"message"`
}

// daemonResponse matches daemon.Response{Code, Message, Data}.
type daemonResponse struct {
	Code    int             `json:"code"`
	Message string          `json:"message"`
	Data    json.RawMessage `json:"data,omitempty"`
}

// DaemonClient talks to the k2 daemon HTTP API on localhost.
type DaemonClient struct {
	Addr string // e.g. "127.0.0.1:1777"
	http *http.Client
}

func NewDaemonClient(addr string) *DaemonClient {
	return &DaemonClient{
		Addr: addr,
		http: &http.Client{Timeout: 5 * time.Second},
	}
}

func (d *DaemonClient) baseURL() string {
	return "http://" + d.Addr
}

// Ping checks if the daemon is reachable (GET /ping, 2s timeout).
func (d *DaemonClient) Ping() error {
	client := &http.Client{Timeout: 2 * time.Second}
	resp, err := client.Get(d.baseURL() + "/ping")
	if err != nil {
		return fmt.Errorf("k2 daemon is not running. Start it with 'k2' or install as a service with 'k2 service install'")
	}
	resp.Body.Close()
	return nil
}

// Up sends the "up" action with a server URL.
func (d *DaemonClient) Up(serverURL string) error {
	body := map[string]any{
		"action": "up",
		"params": map[string]any{
			"config": map[string]any{
				"server": serverURL,
			},
		},
	}
	resp, err := d.postCore(body)
	if err != nil {
		return err
	}
	if resp.Code != 0 {
		return fmt.Errorf("connect failed: %s", resp.Message)
	}
	return nil
}

// Down sends the "down" action.
func (d *DaemonClient) Down() error {
	_, err := d.postCore(map[string]any{"action": "down"})
	return err
}

// Status sends the "status" action and parses the response.
func (d *DaemonClient) Status() (*DaemonStatus, error) {
	resp, err := d.postCore(map[string]any{"action": "status"})
	if err != nil {
		return nil, err
	}
	if len(resp.Data) == 0 {
		return &DaemonStatus{State: "disconnected"}, nil
	}
	var status DaemonStatus
	if err := json.Unmarshal(resp.Data, &status); err != nil {
		return nil, fmt.Errorf("parse status: %w", err)
	}
	return &status, nil
}

func (d *DaemonClient) httpClient() *http.Client {
	if d.http != nil {
		return d.http
	}
	return &http.Client{Timeout: 5 * time.Second}
}

func (d *DaemonClient) postCore(body any) (*daemonResponse, error) {
	data, err := json.Marshal(body)
	if err != nil {
		return nil, fmt.Errorf("marshal: %w", err)
	}
	resp, err := d.httpClient().Post(d.baseURL()+"/api/core", "application/json", bytes.NewReader(data))
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()

	var dr daemonResponse
	if err := json.NewDecoder(resp.Body).Decode(&dr); err != nil {
		return nil, fmt.Errorf("decode daemon response: %w", err)
	}
	return &dr, nil
}
```

```go
package main

import (
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"fmt"
	"net"
	"os"
	"path/filepath"
	"strings"
	"sync"
)

// Session manages authentication tokens and device UDID.
// All exported fields are protected by mu for concurrent access.
type Session struct {
	mu           sync.RWMutex
	AccessToken  string `json:"accessToken"`
	RefreshToken string `json:"refreshToken"`
	IssuedAt     int64  `json:"issuedAt"`
	Email        string `json:"email"`

	dir string // directory for persistence files
}

func NewSession(dir string) *Session {
	return &Session{dir: dir}
}

// SetTokens atomically updates all token fields.
func (s *Session) SetTokens(access, refresh, email string, issuedAt int64) {
	s.mu.Lock()
	s.AccessToken = access
	s.RefreshToken = refresh
	s.Email = email
	s.IssuedAt = issuedAt
	s.mu.Unlock()
}

func (s *Session) sessionPath() string {
	return filepath.Join(s.dir, "mcp-session.json")
}

func (s *Session) udidPath() string {
	return filepath.Join(s.dir, "mcp-udid")
}

// Save persists the session to disk with 0600 permissions.
func (s *Session) Save() error {
	if err := os.MkdirAll(s.dir, 0700); err != nil {
		return fmt.Errorf("create session dir: %w", err)
	}
	data, err := json.MarshalIndent(s, "", "  ")
	if err != nil {
		return fmt.Errorf("marshal session: %w", err)
	}
	return os.WriteFile(s.sessionPath(), data, 0600)
}

// Restore loads a previously saved session. Returns nil if file doesn't exist.
func (s *Session) Restore() error {
	data, err := os.ReadFile(s.sessionPath())
	if os.IsNotExist(err) {
		return nil
	}
	if err != nil {
		return fmt.Errorf("read session: %w", err)
	}
	return json.Unmarshal(data, s)
}

// Clear removes tokens from memory and deletes the session file.
func (s *Session) Clear() {
	s.AccessToken = ""
	s.RefreshToken = ""
	s.IssuedAt = 0
	s.Email = ""
	os.Remove(s.sessionPath())
}

// LoggedIn returns true if an access token is present.
func (s *Session) LoggedIn() bool {
	s.mu.RLock()
	defer s.mu.RUnlock()
	return s.AccessToken != ""
}

// UDID returns the stable device identifier, reading from file first.
// On first run, generates from hostname+MAC and stores to file.
func (s *Session) UDID() (string, error) {
	// Read from file if it exists.
	data, err := os.ReadFile(s.udidPath())
	if err == nil {
		return strings.TrimSpace(string(data)), nil
	}

	// Generate: sha256(hostname + first-non-loopback-MAC)[:16]
	hostname, _ := os.Hostname()
	mac := firstMAC()
	h := sha256.Sum256([]byte(hostname + mac))
	udid := hex.EncodeToString(h[:])[:16]

	// Persist.
	if err := os.MkdirAll(s.dir, 0700); err != nil {
		return udid, fmt.Errorf("create udid dir: %w", err)
	}
	os.WriteFile(s.udidPath(), []byte(udid), 0600)
	return udid, nil
}

// firstMAC returns the MAC address of the first non-loopback network interface.
func firstMAC() string {
	ifaces, err := net.Interfaces()
	if err != nil {
		return "unknown"
	}
	for _, iface := range ifaces {
		if iface.Flags&net.FlagLoopback != 0 {
			continue
		}
		if len(iface.HardwareAddr) > 0 {
			return iface.HardwareAddr.String()
		}
	}
	return "unknown"
}
```

```go
func (app *App) handleLogin(ctx context.Context, input LoginInput) (*mcp.CallToolResult, any, error) {
	udid, err := app.session.UDID()
	if err != nil {
		return errorResult(fmt.Sprintf("failed to get device ID: %v", err)), nil, nil
	}

	body := map[string]any{
		"email":    input.Email,
		"password": input.Password,
		"udid":     udid,
		"remark":   "k2-mcp",
		"platform": "mcp",
	}

	var result struct {
		AccessToken  string `json:"accessToken"`
		RefreshToken string `json:"refreshToken"`
		IssuedAt     int64  `json:"issuedAt"`
	}
	if err := app.center.Post("/api/auth/login/password", body, &result); err != nil {
		return errorResult(fmt.Sprintf("%v", err)), nil, nil
	}

	// Store tokens (thread-safe).
	app.session.SetTokens(result.AccessToken, result.RefreshToken, input.Email, result.IssuedAt)
	app.center.SetToken(result.AccessToken)

	// Persist to disk.
	app.session.Save()

	return successResult(map[string]any{
		"email": input.Email,
	}), nil, nil
}

		},
	}
```

```go
	if !app.session.LoggedIn() {
		return errorResult("not logged in, please call login first"), nil
	}

	var user struct {
		UUID             string `json:"uuid"`
		ExpiredAt        int64  `json:"expiredAt"`
		IsFirstOrderDone bool   `json:"isFirstOrderDone"`
		LoginIdentifies  []struct {
			Type  string `json:"type"`
			Value string `json:"value"`
		} `json:"loginIdentifies"`
		DeviceCount int64 `json:"deviceCount"`
		InviteCode  *struct {
			Code string `json:"code"`
		} `json:"inviteCode"`
	}

	if err := app.center.Get("/api/user", &user); err != nil {
		return app.handleCenterError(err), nil
	}

	// Extract email from loginIdentifies.
	email := ""
	for _, li := range user.LoginIdentifies {
		if li.Type == "email" {
			email = li.Value
			break
		}
	}

	expiresAt := time.Unix(user.ExpiredAt, 0).UTC()
	inviteCode := ""
	if user.InviteCode != nil {
		inviteCode = user.InviteCode.Code
	}

	return successResult(map[string]any{
		"email":           email,
		"plan_expires_at": expiresAt.Format(time.RFC3339),
		"is_active":       user.ExpiredAt > time.Now().Unix(),
		"device_count":    user.DeviceCount,
		"device_limit":    5,
		"invite_code":     inviteCode,
	}), nil
}

// handleCenterError maps CenterError codes to user-friendly MCP error results.
func (app *App) handleCenterError(err error) *mcp.CallToolResult {
	var ce *CenterError
	if errors.As(err, &ce) {
		switch ce.Code {
		case 401:
			return errorResult("not logged in, please call login first")
		case 402:
			return errorResult("subscription expired")
		default:
			return errorResult(ce.Message)
		}
	}
	return errorResult(err.Error())
}
```

```go
import (
	"context"
	"time"

	"github.com/modelcontextprotocol/go-sdk/mcp"

const serverCacheTTL = 5 * time.Minute

// Server is the cached representation of a VPN server.
type Server struct {
	ID                    int     `json:"id"`
	Name                  string  `json:"name"`
	Domain                string  `json:"domain"`
	Country               string  `json:"country"`
	Region                string  `json:"region"`
	TrafficUsagePercent   float64 `json:"traffic_usage_percent"`
	BandwidthUsagePercent float64 `json:"bandwidth_usage_percent"`
	ServerURL             string  `json:"server_url"`
}

type ListServersInput struct{} // no input

func (app *App) registerListServers(server *mcp.Server) {
	mcp.AddTool(server, &mcp.Tool{
		Name:        "list_servers",
		Description: "Get available VPN servers for connection",
	}, func(ctx context.Context, req *mcp.CallToolRequest, input ListServersInput) (*mcp.CallToolResult, any, error) {
		result, err := app.handleListServers(ctx)
		return result, nil, err
	})
}

func (app *App) handleListServers(ctx context.Context) (*mcp.CallToolResult, error) {
	if !app.session.LoggedIn() {
		return errorResult("not logged in, please call login first"), nil
	}

	servers, err := app.fetchServers()
	if err != nil {
		return app.handleCenterError(err), nil
	}

	return successResult(map[string]any{"servers": servers}), nil
}

// fetchServers returns cached servers or fetches from Center API.
func (app *App) fetchServers() ([]Server, error) {
	app.serversMu.RLock()
	if len(app.servers) > 0 && time.Since(app.serversCachedAt) < serverCacheTTL {
		cached := app.servers
		app.serversMu.RUnlock()
		return cached, nil
	}
	app.serversMu.RUnlock()

	// Fetch from Center API.
	var data struct {
		Items []struct {
			ID        int    `json:"id"`
			Domain    string `json:"domain"`
			Name      string `json:"name"`
			ServerURL string `json:"serverUrl"`
			Node      struct {
				Name                  string  `json:"name"`
				Country               string  `json:"country"`
				Region                string  `json:"region"`
				TrafficUsagePercent   float64 `json:"trafficUsagePercent"`
				BandwidthUsagePercent float64 `json:"bandwidthUsagePercent"`
			} `json:"node"`
		} `json:"items"`
	}

	if err := app.center.Get("/api/tunnels/k2v5", &data); err != nil {
		return nil, err
	}

	servers := make([]Server, 0, len(data.Items))
	for _, item := range data.Items {
		name := item.Name
		if name == "" {
			name = item.Node.Name
		}
		servers = append(servers, Server{
			ID:                    item.ID,
			Name:                  name,
			Domain:                item.Domain,
			Country:               item.Node.Country,
			Region:                item.Node.Region,
			TrafficUsagePercent:   item.Node.TrafficUsagePercent,
			BandwidthUsagePercent: item.Node.BandwidthUsagePercent,
			ServerURL:             item.ServerURL,
		})
	}

	app.serversMu.Lock()
	app.servers = servers
	app.serversCachedAt = time.Now()
	app.serversMu.Unlock()

	return servers, nil
}

// findServer looks up a server by ID from the cache.
func (app *App) findServer(id int) *Server {
	app.serversMu.RLock()
	defer app.serversMu.RUnlock()
	for _, s := range app.servers {
		if s.ID == id {
			return &s
		}
	}
	return nil
}

// findServerByDomain looks up a server by domain from the cache.
func (app *App) findServerByDomain(domain string) *Server {
	app.serversMu.RLock()
	defer app.serversMu.RUnlock()
	for _, s := range app.servers {
		if s.Domain == domain {
			return &s
		}
	}
	return nil
}
```

```go
package main

import (
	"context"
	"fmt"

	"github.com/modelcontextprotocol/go-sdk/mcp"
)

type ConnectInput struct {
	ServerID int `json:"server_id" jsonschema:"server ID from list_servers"`
}

func (app *App) registerConnect(server *mcp.Server) {
	mcp.AddTool(server, &mcp.Tool{
		Name:        "connect",
		Description: "Connect to a VPN server",
	}, func(ctx context.Context, req *mcp.CallToolRequest, input ConnectInput) (*mcp.CallToolResult, any, error) {
		result, err := app.handleConnect(ctx, input)
		return result, nil, err
	})
}

func (app *App) handleConnect(ctx context.Context, input ConnectInput) (*mcp.CallToolResult, error) {
	if !app.session.LoggedIn() {
		return errorResult("not logged in, please call login first"), nil
	}

	// Ensure server list is cached.
	if _, err := app.fetchServers(); err != nil {
		return app.handleCenterError(err), nil
	}

	// Resolve server_id → server.
	srv := app.findServer(input.ServerID)
	if srv == nil {
		return errorResult(fmt.Sprintf("server %d not found, call list_servers to see available servers", input.ServerID)), nil
	}

	// Ping daemon.
	if err := app.daemon.Ping(); err != nil {
		return errorResult(err.Error()), nil
	}

	// Send "up" to daemon.
	if err := app.daemon.Up(srv.ServerURL); err != nil {
		return errorResult(err.Error()), nil
	}

	return successResult(map[string]any{
		"state":  "connecting",
		"server": srv.Name,
	}), nil
}
```

```go
	}, func(ctx context.Context, req *mcp.CallToolRequest, input StatusInput) (*mcp.CallToolResult, any, error) {
```

```go
func (app *App) handleStatus(ctx context.Context) (*mcp.CallToolResult, error) {
	status, err := app.daemon.Status()
	if err != nil {
		// Daemon not running = disconnected.
		return successResult(map[string]any{"state": "disconnected"}), nil
	}

	output := map[string]any{
		"state": status.State,
	}

	if status.State == "connected" || status.State == "connecting" || status.State == "reconnecting" {
		if status.UptimeSeconds > 0 {
			output["uptime_seconds"] = status.UptimeSeconds
		}
		// Resolve server name from config.server URL domain.
		if status.Config != nil && status.Config.Server != "" {
			output["server"] = app.resolveServerName(status.Config.Server)
		}
	}

	if status.Error != nil {
		output["error"] = status.Error.Message
		output["error_code"] = status.Error.Code
	}

	return successResult(output), nil
}

// resolveServerName extracts the domain from a k2v5:// URL and looks up
// the friendly name from cached server list. Falls back to raw domain.
func (app *App) resolveServerName(serverURL string) string {
	u, err := url.Parse(serverURL)
	if err != nil {
		return serverURL
	}
	domain := u.Hostname()

	if srv := app.findServerByDomain(domain); srv != nil {
		return srv.Name
	}
	return domain
}
```

- [ ] **Step 3: Add auto-refresh logic to CenterClient at Get/Post level**

The key insight: retry must happen at the `Get()`/`Post()` level, NOT inside `do()`, because `do()` receives a consumed `*http.Request` whose body can't be replayed for POST requests.

Add a `refreshSource` field to `CenterClient`:

```go
// In center_client.go:
type RefreshSource interface {
	GetRefreshToken() string
	UpdateTokens(access, refresh string, issuedAt int64)
}

// Add fields to CenterClient struct:
// refreshSource RefreshSource

func (c *CenterClient) SetRefreshSource(rs RefreshSource) {
	c.mu.Lock()
	c.refreshSource = rs
	c.mu.Unlock()
}

// Modify Get() to retry on 401:
func (c *CenterClient) Get(path string, result any) error {
	err := c.doGet(path, result)
	if c.isCode401(err) && c.tryRefresh() {
		err = c.doGet(path, result)
	}
	return err
}

// Modify Post() to retry on 401 (body bytes still available):
func (c *CenterClient) Post(path string, body any, result any) error {
	data, err := json.Marshal(body)
	if err != nil {
		return fmt.Errorf("marshal body: %w", err)
	}
	err = c.doPost(path, data, result)
	if c.isCode401(err) && c.tryRefresh() {
		err = c.doPost(path, data, result) // replay with same bytes
	}
	return err
}

func (c *CenterClient) isCode401(err error) bool {
	var ce *CenterError
	return errors.As(err, &ce) && ce.Code == 401
}

func (c *CenterClient) tryRefresh() bool {
	c.mu.RLock()
	rs := c.refreshSource
	c.mu.RUnlock()
	if rs == nil {
		return false
	}
	refreshToken := rs.GetRefreshToken()
	if refreshToken == "" {
		return false
	}

	var result struct {
		AccessToken  string `json:"accessToken"`
		RefreshToken string `json:"refreshToken"`
		IssuedAt     int64  `json:"issuedAt"`
	}
	body, _ := json.Marshal(map[string]string{"refreshToken": refreshToken})
	if err := c.doPost("/api/auth/refresh", body, &result); err != nil {
		return false
	}

	// Update both CenterClient token AND session.
	c.SetToken(result.AccessToken)
	rs.UpdateTokens(result.AccessToken, result.RefreshToken, result.IssuedAt)
	return true
}
```

Note: `tryRefresh` calls `c.SetToken(result.AccessToken)` to update the in-memory token on `CenterClient`, AND calls `rs.UpdateTokens()` to persist to disk. Both must happen for subsequent requests to use the new token.
